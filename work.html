Python 学习笔记

基础语法

1. 变量和数据类型


# 变量赋值
x = 5
y = "Hello, World!"

# 数据类型
integer_num = 10
float_num = 3.14
string_text = "Python"
boolean_value = True
list_example = [1, 2, 3]
tuple_example = (1, 2, 3)
dict_example = {"name": "John", "age": 30}

2. 条件语句


if x > 0:
    print("x is positive")
elif x < 0:
    print("x is negative")
else:
    print("x is zero")

3. 循环


# For 循环
for i in range(5):
    print(i)

# While 循环
count = 0
while count < 5:
    print(count)
    count += 1

4. 函数


def greet(name):
    return f"Hello, {name}!"

print(greet("Alice"))

基础语法

1. 变量和数据类型


#include <iostream>
#include <string>

int main() {
    // 变量赋值
    int x = 5;
    std::string y = "Hello, World!";

    // 数据类型
    int integer_num = 10;
    float float_num = 3.14f;
    std::string string_text = "C++";
    bool boolean_value = true;
    
    return 0;
}

2. 条件语句


if (x > 0) {
    std::cout << "x is positive" << std::endl;
} else if (x < 0) {
    std::cout << "x is negative" << std::endl;
} else {
    std::cout << "x is zero" << std::endl;
}

3. 循环


// For 循环
for (int i = 0; i < 5; i++) {
    std::cout << i << std::endl;
}

// While 循环
int count = 0;
while (count < 5) {
    std::cout << count << std::endl;
    count++;
}

4. 函数


#include <string>

std::string greet(const std::string& name) {
    return "Hello, " + name + "!";
}

int main() {
    std::cout << greet("Alice") << std::endl;
    return 0;
}
