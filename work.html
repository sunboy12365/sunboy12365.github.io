高级特性

1. 装饰器


import time

def timer(func):
    def wrapper(*args, **kwargs):
        start = time.time()
        result = func(*args, **kwargs)
        end = time.time()
        print(f"函数 {func.__name__} 运行时间：{end - start}秒")
        return result
    return wrapper

@timer
def slow_function():
    time.sleep(2)
    print("这是一个耗时的函数")

slow_function()

2. 生成器


def fibonacci(n):
    a, b = 0, 1
    for _ in range(n):
        yield a
        a, b = b, a + b

for num in fibonacci(10):
    print(num)

高级特性

1. 类和对象


class Animal {
protected:
    std::string name;

public:
    Animal(const std::string& name) : name(name) {}
    virtual void makeSound() = 0;
};

class Dog : public Animal {
public:
    Dog(const std::string& name) : Animal(name) {}
    void makeSound() override {
        std::cout << name << "汪汪叫" << std::endl;
    }
};

int main() {    
    Dog dog("旺财");
    dog.makeSound();  // 输出：旺财汪汪叫
    return 0;
}

2. 模板


template <typename T>
T max(T a, T b) {
    return (a > b) ? a : b;
}

int main() {
    std::cout << max(3, 7) << std::endl;  // 输出：7
    std::cout << max(3.14, 2.72) << std::endl;  // 输出：3.14
    return 0;
}
